2306 公司命名

题目

  • 给你一个字符串数组 ideas 表示在公司命名过程中使用的名字列表。公司命名流程如下：
  1. 从 ideas 中选择 2 个 不同 名字，称为 ideaA 和 ideaB 。
  2. 交换 ideaA 和 ideaB 的首字母。
  3. 如果得到的两个新名字 都 不在 ideas 中，那么 ideaA ideaB（串联 ideaA 和 ideaB ，中间用一个空格分隔）是一个有效的公司名字。
  4. 否则，不是一个有效的名字。
  5. 返回 不同 且有效的公司名字的数目

示例

输入：ideas = [“coffee”,“donuts”,“time”,“toffee”] 输出：6 解释：下面列出一些有效的选择方案：

  • (“coffee”, “donuts”)：对应的公司名字是 “doffee conuts” 。
  • (“donuts”, “coffee”)：对应的公司名字是 “conuts doffee” 。
  • (“donuts”, “time”)：对应的公司名字是 “tonuts dime” 。
  • (“donuts”, “toffee”)：对应的公司名字是 “tonuts doffee” 。
  • (“time”, “donuts”)：对应的公司名字是 “dime tonuts” 。
  • (“toffee”, “donuts”)：对应的公司名字是 “doffee tonuts” 。 因此，总共有 6 个不同的公司名字。

下面列出一些无效的选择方案：

  • (“coffee”, “time”)：在原数组中存在交换后形成的名字 “toffee” 。
  • (“time”, “toffee”)：在原数组中存在交换后形成的两个名字。
  • (“coffee”, “toffee”)：在原数组中存在交换后形成的两个名字。

题解

方案一

  • 代码
class Solution {
public:
    long long distinctNames(vector<string>& ideas) {
        set<string> new_ideas[26];
        for(string idea: ideas) {
            new_ideas[idea[0] - 'a'].insert(idea.substr(1)); // 将所有子串放到首字母索引的集合里
        }
        long long num = 0;
        for(char c1 = 'a'; c1 <= 'z'; c1++) { // 遍历所有首字母索引的集合
            long long before_num1 = new_ideas[c1 - 'a'].size();
            long long before_num2 = 0;
            if(before_num1) {
                for(char c2 = c1 + 1; c2 <= 'z'; c2++) { // 两两对照
                    before_num2 = new_ideas[c2 - 'a'].size();
                    if(before_num2)
                    {
                        set<string> name_sum = new_ideas[c1 - 'a']; // 合并两个集合
                        name_sum.insert(new_ideas[c2 - 'a'].begin(), new_ideas[c2 - 'a'].end());
                        long long new_value = 2 * name_sum.size() - before_num1 - before_num2 + num;
                        num = new_value;
                    }
                }
            }
        }
        return num;
    }
};
  • 提交结果
    • 19/81通过
    • 答案错误
  • 分析
    • 思路
      • 将所有idea按首字母分组，取除首字母外的子串构成多个集合，多个集合构成数组
      • 将集合两两合并，根据差值可计算出重复子串的数目
      • 用合集的元素数目 - (seta.size() + setb.size()- 合集的元素数目)即为这两组所贡献的idea
    • 问题
      • 计算方法错误，不能平均一个不重复的名字贡献一个新名字

class Solution {
public:
    long long distinctNames(vector<string>& ideas) {
        set<string> new_ideas[26];
        for(string idea: ideas) {
            new_ideas[idea[0] - 'a'].insert(idea.substr(1)); // 将所有子串放到首字母索引的集合里
        }
        long long num = 0;
        for(char c1 = 'a'; c1 <= 'z'; c1++) { // 遍历所有首字母索引的集合
            long long before_num1 = new_ideas[c1 - 'a'].size();
            long long before_num2 = 0;
            if(before_num1) {
                for(char c2 = c1 + 1; c2 <= 'z'; c2++) { // 两两对照
                    before_num2 = new_ideas[c2 - 'a'].size();
                    if(before_num2)
                    {
                        set<string> name_sum = new_ideas[c1 - 'a']; // 合并两个集合
                        name_sum.insert(new_ideas[c2 - 'a'].begin(), new_ideas[c2 - 'a'].end());
                        long long differ = before_num1 + before_num2 - name_sum.size();
                        num += (before_num1 - differ) * (before_num2 - differ) * 2;
                    }
                }
            }
        }
        return num;
    }
};

方案三

class Solution {
public:
    long long distinctNames(vector<string>& ideas) {
        int multiple[26][26];
        for(int i = 0; i < 26; i++) {
            for(int j = 0; j < 26; j++) {
                multiple[i][j] = 0;
            }
        }
        vector<string> new_ideas[26];
        for (long long i = 0; i < ideas.size(); i++) {
            new_ideas[ideas[i][0] - 'a'].push_back(ideas[i].substr(1)); 
        } // 创建以首字母为索引的数组
        long long half_new_num = 0;
        for (int i = 0; i < 26; i++) {
            long long a = new_ideas[i].size();
            if(a) {
                for (int j = i+1; j < 26; j++) {
                    long long b = new_ideas[j].size();
                    half_new_num += a * b;
                }
            }
        }
        
        for (int c1 = 0; c1 < 26; c1++) {
            // 遍历当前字母为索引的所有字符串
            for (long long i = 0; i < new_ideas[c1].size(); i++) {
                cout << half_new_num << endl;
                
                // 遍历所有后续字母为索引的字符串
                for (int c2 = c1 + 1; c2 < 26; c2++) {
                    for (long long j = 0; j < new_ideas[c2].size(); j++) {
                        if(new_ideas[c1][i] == new_ideas[c2][j]) {
                            cout << "c1 = " << c1 << ", c2 = " << c2 << endl;
                            if (!multiple[c1][c2])
                                multiple[c1][c2]++;
                            else
                                multiple[c1][c2] += 2;
                            half_new_num = half_new_num - new_ideas[c2].size() - new_ideas[c1].size() + multiple[c1][c2];
                        }
                    }
                }
            }
        }
        
        return half_new_num * 2;
    }
};
  • 提交结果
    • 84/89通过
    • 超出时间限制
  • 分析
    • 按首字母分组，放到二维数组中，以首字母为索引，存储除首字母之外的子串
    • 先计算所有的新名字的个数
    • 再遍历所有子串，减去相同的子串造成的新名字的减少量
    • 相减得到最后结果
    • 其实和方案二一样，只是把set自动去掉重复子串换成了vector手动比较

方案四(力扣官方题解)

class Solution {
public:
    long long distinctNames(vector<string>& ideas) {
        unordered_map<char, unordered_set<string>> names;
        for (const string& idea: ideas) {
            names[idea[0]].insert(idea.substr(1, idea.size() - 1));
        }

        auto get_intersect_size = [](const unordered_set<string>& a, const unordered_set<string>& b) -> size_t {
            size_t ans = 0;
            for (const string& s: a) {
                if (b.count(s)) {
                    ++ans;
                }
            }
            return ans;
        };

        long long ans = 0;
        for (auto&& [pre_a, set_a]: names) {
            for (auto&& [pre_b, set_b]: names) {
                if (pre_a == pre_b) {
                    continue;
                }
                int intersect = get_intersect_size(set_a, set_b);
                ans += static_cast<long long>(set_a.size() - intersect) * (set_b.size() - intersect);
            }
        }
        return ans;
    }
};
