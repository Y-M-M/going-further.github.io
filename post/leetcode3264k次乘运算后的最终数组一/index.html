题目

给你一个整数数组 nums ，一个整数 k 和一个整数 multiplier 。
你需要对 nums 执行 k 次操作，每次操作中：
找到 nums 中的 最小 值 x ，如果存在多个最小值，选择最 前面 的一个。
将 x 替换为 x * multiplier 。
请你返回执行完 k 次乘运算之后，最终的 nums 数组。

示例

示例一

输入：nums = [2,1,3,5,6], k = 5, multiplier = 2
输出：[8,4,6,5,6]
解释

示例二

输入：nums = [1,2], k = 3, multiplier = 4
输出：[16,8]
解释：

提示

  • 1 <= nums.length <= 100
  • 1 <= nums[i] <= 100
  • 1 <= k <= 10
  • 1 <= multiplier <= 5

题解

方法一

class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        for (int i = 0; i < k; i++)
        {
            int pos = 0;
            int minnum = nums[0];
            for (int j = 1; j < nums.size(); j++)
            {
                if (nums[j] < minnum)
                {
                    pos = j;
                    minnum = nums[j];
                }
            }
            nums[pos] *= multiplier;
        }
        return nums;
    }
};

方法二

class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        auto cmp = [&nums](int left, int right) {return nums[left] > nums[right];};
        priority_queue<int, vector<int>, decltype(cmp)> id(cmp);

        for (int i = 0; i < nums.size(); i++)
        {
            id.push(i);
        }
        for (int i = 0; i < k; i++)
        {
            int updatetarget = id.top();
            id.pop();
            nums[updatetarget] *= multiplier;
            id.push(updatetarget);
        }
        return nums;
    }
};
  • 提交结果
    • 一：584/373个通过的测试样例
  • 分析
    • 优先级队列
    • 错误原因
      • 先pop后push，破坏了排序的稳定性，每次取出的最小元素不是按顺序排列的第一个

方法三

class Solution {
public:
    long long quickMul(long long x, long long y, long long m)
    {
        long long res = 1;
        while (y)
        {
            if (y & 1) {
                res = (res * x) % m;
            }
            y >>= 1;
            x = (x * x) % m;
        }
        return res;
    }
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        if (multiplier == 1)
        {
            return nums;
        }
        long long n = nums.size(), m = 1e9 + 7;
        long long mx = *max_element(nums.begin(), nums.end()); // 取最大值
        vector<pair<long long, int>> v(n);
        for (int i = 0; i < n; i++)
        {
            v[i] = {nums[i], i};
        }
        make_heap(v.begin(), v.end(), greater<>());
        for (; v[0].first < mx && k; k--)
        {
            pop_heap(v.begin(), v.end(), greater<>());
            v.back().first *= multiplier;
            push_heap(v.begin(), v.end(), greater<>());
        }
        sort(v.begin(), v.end());
        for (int i = 0; i < n; i++)
        {
            int t = k / n + (i < k % n);
            nums[v[i].second] = ((v[i].first % m) * quickMul(multiplier, t, m)) % m;
        }
        return nums;
    }
};
  • 提交结果
    • 一：通过
    • 二：通过
  • 分析
    • 通过pair的排序规则来进行堆排序，先比较first，first相同的时候在比较second
    • 对过程进行优化
      • 如果nums中所有数都被更新过了，那么此后更新会从小到大把nums中的所有数都*k
      • 这个过程可直接遍历进行，无需再进行堆排序
