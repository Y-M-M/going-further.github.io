leetcode 743 网络延迟时间

题目

有 n 个网络节点，标记为 1 到 n。
给你一个列表 times，表示信号经过 有向 边的传递时间。 times[i] = $(u_i, v_i, w_i)$，其中 $u_i$ 是源节点，$v_i$ 是目标节点， $w_i$ 是一个信号从源节点传递到目标节点的时间。
现在，从某个节点 K 发出一个信号。需要多久才能使所有节点都收到信号？如果不能使所有节点收到信号，返回 -1 。

提示

  • 1 <= k <= n <= 100
  • 1 <= times.length <= 6000
  • times[i].length == 3
  • 1 <= $u_i, v_i$ <= n
  • $u_i$ != $v_i$
  • 0 <= $w_i$ <= 100
  • 所有 (ui, vi) 对都 互不相同（即，不含重复边）

题解

方法一

class Solution {
public:
    int networkDelayTime(vector<vector<int>>& times, int n, int k) {
        vector<vector<int>> next(n + 1); // 记录每个节点的出边存放的位置
        for (int i = 0; i < times.size(); i++) {
            next[times[i][0]].push_back(i);
        }
        
        vector<bool> is_visited(n + 1, false);
        int root = k;
        is_visited[k] = true;
        vector<int> ans(n + 1);
        ans[root] = 0; 
        
        auto cmp = [&](const int &u, const int &v) {
            return ans[times[u][0]] + times[u][2] > ans[times[v][0]] + times[v][2];
        };
        priority_queue<int, vector<int>, decltype(cmp)> pq(cmp);
        int newnode = 0;
        int res = 0;
        
        do {
            for (int i = 0; i < next[root].size(); i++) {
                pq.emplace(next[root][i]);
            }
            
            if (pq.empty()) break; // 检查队列是否为空
            
            newnode = pq.top(); // 加入的边在times中的位置
            pq.pop();
            
            // 找到下一个未访问的节点
            while (!pq.empty() && is_visited[times[newnode][1]]) {
                newnode = pq.top();
                pq.pop();
            }
            
            // 如果pq为空，说明没有未访问的节点，提前退出
            if (is_visited[times[newnode][1]]) break;
            ans[times[newnode][1]] = ans[times[newnode][0]] + times[newnode][2];
            res = ans[times[newnode][1]];
            root = times[newnode][1];
            is_visited[root] = true; // 标记当前节点为已访问
        } while (true);
        
        // 检查是否所有节点都被访问过
        for (int i = 1; i <= n; i++) {
            if (!is_visited[i])
                return -1;
        }
        return res;
    }
};

    • 类比Djkstra算法求最短路径
    • 不在更新时进行松弛操作，而是每次都将剩下节点中离起始节点最近的节点加入visited中
    • 最后加入vistited的一定是离起始节点最远的节点，更新的距离即为离最远节点的最小距离，即为最大传播时间
    • 优先级队列维护最小堆

方法二

class Solution {
public:
    int networkDelayTime(vector<vector<int>> &times, int n, int k) {
        const int inf = INT_MAX / 2;
        vector<vector<int>> g(n, vector<int>(n, inf));
        for (auto &t : times) {
            int x = t[0] - 1, y = t[1] - 1; // x与y分别代表边的两端的节点
            g[x][y] = t[2]; // 二维数组g存储边的权重
        }

        vector<int> dist(n, inf); // 存储距离
        dist[k - 1] = 0; // 跟节点到自己的距离为0
        vector<int> used(n); // 检查节点有么有被使用过
        for (int i = 0; i < n; ++i) {
            int x = -1;
            for (int y = 0; y < n; ++y) { // 找出distance最小的，没有使用过的x
                if (!used[y] && (x == -1 || dist[y] < dist[x])) {
                    x = y;
                }
            }
            used[x] = true; // 将x标记为已经使用过
            for (int y = 0; y < n; ++y) {
                dist[y] = min(dist[y], dist[x] + g[x][y]); // 看新加入的x节点能否使其他节点的距离变小
            }
        }

        int ans = *max_element(dist.begin(), dist.end());
        return ans == inf ? -1 : ans;
    }
};
