leetcode 3250 单调数组对的数目一

题目

给你一个长度为 n 的 正 整数数组 nums 。
如果两个 非负 整数数组 (arr1, arr2) 满足以下条件，我们称它们是 单调 数组对：

  • 两个数组的长度都是 n 。
  • arr1 是单调 非递减 的，换句话说 arr1[0] <= arr1[1] <= … <= arr1[n - 1] 。
  • arr2 是单调 非递增 的，换句话说 arr2[0] >= arr2[1] >= … >= arr2[n - 1] 。
  • 对于所有的 0 <= i <= n - 1 都有 arr1[i] + arr2[i] == nums[i] 。
    请你返回所有 单调 数组对的数目。 由于答案可能很大，请你将它对 109 + 7 取余 后返回。

示例

示例一

输入：nums = [2,3,2]
输出：4
解释：
单调数组对包括：

  1. ([0, 1, 1], [2, 2, 1])
  2. ([0, 1, 2], [2, 2, 0])
  3. ([0, 1, 2], [2, 2, 0])
  4. ([1, 2, 2], [1, 1, 0])

示例二

输入：nums = [5,5,5,5]
输出：126

提示

  • 1 <= n == nums.length <= 2000
  • 1 <= nums[i] <= 50

题解

方法一

class Solution {
public:
    int countOfPairs(vector<int>& nums) {
        vector<vector<int>> dp(nums.size(), vector<int>(51, 0));
        int mod = 1e9 + 7;

        for (int i = 0; i <= nums[0]; i++)
        {
            dp[0][i] = 1;
        }

        for (int i = 1; i < nums.size(); i++)
        {
            for (int v2 = 0; v2 <= nums[i]; v2++)
            {
                for (int v1 = 0; v1 <= v2; v1++)
                {
                    if (nums[i] - v2 <= nums[i - 1] - v1 && nums[i] - v2 >= 0)
                        dp[i][v2] = (dp[i][v2] + dp[i - 1][v1]) % mod;
                }
            }
        }

        int ans = 0;
        for (int i = 0; i <= nums.back(); i++)
        {
            ans = (ans + dp[nums.size() - 1][i]) % mod;
        }
        return ans;
    }
};
  • 提交结果
    • 通过
  • 分析
    • 动态规划
    • dp[i][j] 表示当 $arr_1$[i] = j 时，前 i + 1 个元素组成的单调数组的数目
    • 因为 $arr_1$[0] 可以为 0 到 nums[0] 之间的任意数，初始化 dp[0][j]=1，其中 j 小于 nums[0]，其它初始化为零。
    • dp[i][$v_2$] = $\sum_{v1<=v2, nums[i - 1] - v1 ≥ nums[i] - v2 ≥ 0}$(dp[i - 1][$v_1$])

方法二

class Solution {
public:
    int countOfPairs(vector<int>& nums) {
        vector<vector<int>> dp(nums.size(), vector<int>(51, 0));
        int mod = 1e9 + 7;

        for (int i = 0; i <= nums[0]; i++)
        {
            dp[0][i] = 1;
        }

        for (int i = 1; i < nums.size(); i++)
        {
            int d = max(0, nums[i] - nums[i - 1]);
            for (int j = d; j <= nums[i]; j++)
            {
                if (j == 0) {
                    dp[i][j] = dp[i - 1][j - d];
                }
                else
                {
                    dp[i][j] = (dp[i][j - 1] + dp[i - 1][j - d]) % mod;
                }
            }
        }

        int ans = 0;
        for (int i = 0; i <= nums.back(); i++)
        {
            // cout << "i = " << i << ", dp[nums.size() - 1][i] = " << dp[nums.size() - 1][i] << endl;
            ans = (ans + dp[nums.size() - 1][i]) % mod;
        }
        return ans;
    }
};
  • 提交结果
    • 通过
  • 分析
    • 动态规划
    • 在方法一的基础上改进
    • dp[i][j] = dp[i][j - 1] + dp[i - 1][j - d](j ≥ d, d = max(0, nums[i] - nums[i - 1]))
