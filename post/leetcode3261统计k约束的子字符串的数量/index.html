leetcode 3261 统计K约束的子字符串数量二

题目

给你一个 二进制 字符串 s 和一个整数 k。 另给你一个二维整数数组 queries ，其中 queries[i] = [li, ri] 。 如果一个 二进制字符串 满足以下任一条件，则认为该字符串满足 k 约束：

  • 字符串中 0 的数量最多为 k。
  • 字符串中 1 的数量最多为 k。 返回一个整数数组 answer ，其中 answer[i] 表示 s[li..ri] 中满足 k 约束 的 子字符串 的数量。

示例

示例一

输入：s = “0001111”, k = 2, queries = [[0,6]] 输出：[26] 解释： 对于查询 [0, 6]， s[0..6] = “0001111” 的所有子字符串中，除 s[0..5] = “000111” 和 s[0..6] = “0001111” 外，其余子字符串都满足 k 约束。

示例二

输入：s = “010101”, k = 1, queries = [[0,5],[1,4],[2,3]] 输出：[15,9,3] 解释： s 的所有子字符串中，长度大于 3 的子字符串都不满足 k 约束。

提示

  • 1 <= s.length <= $10^5$
  • s[i] 是 ‘0’ 或 ‘1’
  • 1 <= k <= s.length
  • 1 <= queries.length <= $10^5$
  • queries[i] == $[l_i, r_i]$
  • $0 <= l_i <= r_i < s.length$
  • 所有查询互不相同

题解

方法一

class Solution {
public:
    vector<long long> countKConstraintSubstrings(string s, int k, vector<vector<int>>& queries) {
        int n = s.length();
        vector<int> pre_sum0(n); // 前缀和数组，记录前缀0的个数(前缀1的个数可以由总数减去前缀0的个数获得)
        if (s[0] == '0')
        {
            pre_sum0[0] = 1;
        }
        else
        {
            pre_sum0[0] = 0;
        }
        for (int i = 1; i < n; i++)
        {
            if (s[i] == '0')
            {
                pre_sum0[i] = pre_sum0[i - 1] + 1;
            }
            else 
                pre_sum0[i] = pre_sum0[i - 1];
        }
        vector<vector<long long>> sub_string_num(n); // 记录从i开始到j结束的子串中的K约束子串数目
        for (int i = 0; i < n; i++) {
            sub_string_num[i].resize(n - i);
        }

        for (int i = 0; i < n; i++) { // 初始化sub_string_num数组，当k不为0的时候，
            if (k)
                sub_string_num[i][0] = 1;
        }

        for (int len = 1; len < n; len++) 
        {
            for (int l = 0; l < n - len; l++)
            {
                long long now = 0;
                int r = l + len;
                now += sub_string_num[l][(l + r) / 2 - l];
                now += sub_string_num[(l + r) / 2 + 1][r - ((l + r) / 2 + 1)];
                int i = (l + r) / 2, j = i + 1;
                i--;
                while (i >= l) {
                    for (int t = j; t <= r; t++)
                    {
                        int num0 = pre_sum0[t] - pre_sum0[i];
                        if (num0 <= k || (t - i) - num0 <= k)
                        {
                            now++;
                        }
                        else
                            break;
                    }
                    i--;
                }
                for (int t = j; t <= r; t++) 
                {
                    int num0 = 0;
                    if (l - 1 < 0)
                        num0 = pre_sum0[t];
                    else 
                        num0 = pre_sum0[t] - pre_sum0[l - 1];
                    if (num0 <= k || (t - l + 1) - num0 <= k)
                    {
                        now++;
                    }
                    else 
                        break;
                }
                sub_string_num[l][r - l] = now;
            }
        }

        vector<long long> ans;
        for (int i = 0; i < queries.size(); i++)
        {
            int l = queries[i][0], r = queries[i][1];
            ans.push_back(sub_string_num[l][r - l]);
        }

        return ans;
    }
};

  • 思路
    • 类似动态规划/分治法的解题方法
    • 区间[l, r]的所有K约束的子串的数量来源
      • [l, k]中满足K约束的子串数量
      • [k + 1, r]中满足K约束子串的数量
      • 跨两个区间的K约束子串数量
    • 使用sub_string_num记录，区间长度从1到n，依次计算所有子串的满足K约束的子串数目
    • 细节
      • k取(l + r)/2
      • 判断一个子段是否满足要求：根据前缀和数组
      • sub_string_num初始化：对于所有长度为1的子串，满足K约束的子串的数目为1
      • 为节省空间，将sub_string_num数组压缩存储，只存储对角线以上的元素(但空间还是会爆掉…)

方法二

class Solution {
public:
    vector<long long> countKConstraintSubstrings(string s, int k, vector<vector<int>>& queries) {
        int n = s.size();
        vector<int> count(2, 0); // 统计0和1的数目
        vector<int> right(n, n);
        vector<long long> prefix(n + 1, 0);
        int i = 0;
        for (int j = 0; j < n; ++j) {
            count[s[j] - '0']++;
            while (count[0] > k && count[1] > k) {
                count[s[i] - '0']--;
                right[i] = j; // 将i的右极限记为j
                i++;
            }
            prefix[j + 1] = prefix[j] + j - i + 1; // prefix[j + 1]右端点在j的所有满足条件的子串数目
        }

        vector<long long> res;
        for (int k = 0; k < queries.size(); k++)
        {
            int l = queries[k][0], r = queries[k][1];
            int j = min(right[l], r + 1);
            long long part1 = (long long) (j - l + 1) * (j - l) / 2;
            long long part2 = prefix[r + 1] - prefix[j];
            res.push_back(part1 + part2);
        }
        return res;
    }
};
