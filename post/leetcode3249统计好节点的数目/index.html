leetcode 3249 统计好节点的数目

题目

现有一棵 无向 树，树中包含 n 个节点，按从 0 到 n - 1 标记。树的根节点是节点 0 。给你一个长度为 n - 1 的二维整数数组 edges，其中 edges[i] = [ai, bi] 表示树中节点 ai 与节点 bi 之间存在一条边。 如果一个节点的所有子节点为根的 子树 包含的节点数相同，则认为该节点是一个 好节点。 返回给定树中 好节点 的数量。 子树 指的是一个节点以及它所有后代节点构成的一棵树。

示例

示例一

输入：edges = [[0,1],[0,2],[1,3],[1,4],[2,5],[2,6]] 输出：7 说明

示例二

输入：edges = [[0,1],[1,2],[2,3],[3,4],[0,5],[1,6],[2,7],[3,8]] 输出：6 说明：

示例三

输入：edges = [[0,1],[1,2],[1,3],[1,4],[0,5],[5,6],[6,7],[7,8],[0,9],[9,10],[9,12],[10,11]] 输出：12 解释：

提示

  • 2 <= n <= $10^5$
  • edges.length == n - 1
  • edges[i].length == 2
  • 0 <= $a_i$, $b_i$ < n
  • 输入确保 edges 总表示一棵有效的树。

题解

方法一

class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            cnt += dfs(tree[root][i], tree, sum);
        }
        sum[root] = cnt;
        return cnt;
    }

    void buildit(vector<vector<int>>& tree, const int& root) // 将连接关系缩减为父子关系
    {
        for (int i = 0; i < tree[root].size(); i++)
        {
            int son = tree[root][i];
            for (int j = 0; j < tree[son].size(); j++)
            {
                if (tree[son][j] == root)
                    tree[son].erase(tree[son].begin() + j);
            }
            buildit(tree, son);
        }
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        buildit(tree, 0); // 将连接关系缩减为父子关系
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum);
        for (int i = 0; i <= n; i++) // 遍历所有节点，判断是不是好节点并统计数目
        {
            int isgood = 1;
            for (int j = 1; j < tree[i].size(); j++)
            {
                if (sum[tree[i][j]] != sum[tree[i][0]])
                {
                    isgood = 0;
                    break;
                }
            }
            ans += isgood;
        }
        return ans;
    }
};
  • 提交结果
    • 通过
  • 分析
    • 时间复杂度O($n^2$)
    • 空间复杂度O(n)
    • 思路
      • 深度优先搜索统计子树的节点数
      • 因为所给的连接关系不会隐含父子关系，需要从根节点开始自己判断，所以先建立连接关系再缩减为父子关系

方法二

class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum, const int& fa, int& ans) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        int num = 0;
        int isgood = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            if (tree[root][i] != fa)
            {
                int add_num = dfs(tree[root][i], tree, sum, root, ans);
                if (!num) num = add_num;
                else {
                    if (add_num != num)
                        isgood = 0;
                }
                cnt += add_num;
            } 
        }
        if (isgood) ans++;
        sum[root] = cnt;
        return cnt;
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum, -1, ans);
        return ans;
    }
};
  • 提交结果
    • 通过
  • 分析
    • 在方法一的基础上改进
      • dfs的同时统计好节点的数目
      • 不手动删除冗余的连接关系，只要在dfs的时候加个判断，不要访问父节点就可以了
    • 时间复杂度O(n)
    • 空间复杂度O(n)
